"new": function(prefix) {
      prefix || (prefix = "");
      return {
        get: function(key) {
          return retrieve("" + prefix + "_" + key);
        },
        set: function(key, value) {
          return store("" + prefix + "_" + key, value);
        },
        put: function(key, value) {
          return store("" + prefix + "_" + key, value);
        }
      };
    }
  };
})();
